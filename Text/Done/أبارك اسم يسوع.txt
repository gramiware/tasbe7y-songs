(يسوع 4)

(يسوع 4)

حدثوا دوماً باسم يسوع
حمدوه بين الشـــــــعوب
(يسوع 4)
حقه دوماً إلى الآبــــــــاد

طوبى لعارفي اسم يسوع
يتكلون دوما عليــــــــه

طماننا إلى الآبــــــــــــاد

يهوه خلاصي اسم يسوع
كل ما في يبـــــــــارك
(يسوع 4)
يهوه ترنيمتي إلى الآبـاد

كرامة لاسم يسوع
في عرش الله وسط الكنيسة
(يسوع 4)
كلي المجد إلى الآبـــــــــاد

لا مثل له اسم يسوع
في الجبروت بين الأسماء
(يسوع 4)
ليتبارك إلى الآبـــــــــــاد

ما أمجد اسم يسوع
في كل الأرض وفي السماء
(يسوع 4)
مجدُ الأمجاد إلى الآبـــاد

نارُ هو اسم يسوع
على الشيطان وعلى الأعداء
(يسوع 4)
نور ونار إلى الابــــــــاد

سيُعرف اسم يسوع
قدام الشمس في كل الأرض
(يسوع 4)
سبح التسبيح إلى الآبــاد

عجيب هو اسم يسوع
وعظيمٌ جداً في القـــــدرة
(يسوع 4)
علم الأعلام إلى الآبـــــاد

فوق الأسماء اسم يسوع
فوق الجميع فوق السماء
(يسوع 4)
فهو العلي إلى الآبـــــــاد

صخرتي هو اسم يسوع
برج حصين وهو سلاحي
(يسوع 4)
صخرة خلاصي إلى الآباد

قلبي ولحمي لاسم يسوع
هما يهتفان ويسـبحان
(يسوع 4)
قدوس الله إلى الآبـــــــاد

ربوات ربوات لاسم يسوع
وألوف ألوف يمجدون
(يسوع 4)
رب الأرباب إلى الآبــــاد

شفتاي تسبح اسم يسوع
شكراً أقدم شــــــهادتي
(يسوع 4)
شَهدُ الشفاه إلى الابــــاد

تاج البهاء اسم يسوع
تجثو له كل الخلائــــــــق
(يسوع 4)
تقديسنا إلى الآبــــــــــاد
